using namespace std;

int w[101];

int main(){
	freopen("lazy_loading.txt","r",stdin);
	freopen("answerB.txt","w",stdout);
	int t;sd(t);
	for(int tt=1;tt<=t;tt++){
		int n;sd(n);
		for(int i=1;i<=n;i++){
			sd(w[i]);
			w[i]*=-1;
		}
		sort(w+1,w+n+1);
		for(int i=1;i<=n;i++) w[i]*=-1;
		int ans=0;
		int r=n;
		for(int i=1;i<=n;i++){
			int k=50/w[i];
			if(50%w[i]) ++k;
			// you need k-1 items 
			int p=k-1; 
			if(r-p+1>i){
				++ans;
				r=r-p;
			}else break;
		}
		printf("Case #%d: ",tt);
		pd(ans);
	}
}
